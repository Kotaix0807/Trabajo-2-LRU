257void agregar_dato(LRUcache *cache, char val) {
258 Position encontrado = FindNode(cache->main, val);
259
260 if(encontrado) {
261 usar_dato(cache, val);
262 return;
263 }
264
265 int count = 0;
266 Position tmp = cache->main->Next;
267 while(tmp) { count++; tmp = tmp->Next; }
268
269 if(count >= cache->max_cache) eliminar_menos_usado(cache);
270
271 Position nuevo = malloc(sizeof(struct Node));
272 if (!nuevo) {
273 perror("Error de asignacion de memoria");
274 exit(EXIT_FAILURE);
275 }
276
277 nuevo->Element = val;
278 // Insercion al inicio (nuevo MRU)
279 nuevo->Next = cache->main->Next;
280 cache->main->Next = nuevo;
281 cache->primero = nuevo;
282
283 // Si la cache estaba vacia, este es el LRU tambien
284 if(!cache->ultimo) {
285 cache->ultimo = nuevo;
286 }

294void usar_dato(LRUcache *cache, char val) {
295 Position prev = cache->main;
296 Position actual = cache->main->Next;
297
298 // Buscar el nodo y su predecesor
299 while(actual && actual->Element != val) {
300 prev = actual;
301 actual = actual->Next;
302 }
303
304 if(!actual) return;
305
306 // Si ya es el primero (MRU)
307 if(prev == cache->main) {
308 printf("Data element %c used (already MRU)\n", val);
309 return;
310 }
311
312 // Si el nodo actual era el LRU, el predecesor es el nuevo LRU
313 if (actual == cache->ultimo) {
314 cache->ultimo = prev;
315 }
316
317 // Quitar de la posicion actual
318 prev->Next = actual->Next;
319
320 // Mover al frente (MRU)
321 actual->Next = cache->main->Next;
322 cache->main->Next = actual;
323 cache->primero = actual;
324
325 printf("Data element %c used\n", val);
326}
